159 """All the nodes we can position the object at in order to

162 nodes = set() 

163 for x in range(0, self.window_width - self.width, grid_spacing):

169 def set_perimeter(self) -> None:

171 self.perimeter_x = [

178 def set_nodes(self) -> None: 

179 """Set coords used by object as non-traversable or set the 

180 associated movement cost if we are in grass or water. 

181 """ 

182 for x in self.perimeter_x: 

183 for y in self.perimeter_y: 

184 if self.name == 'grass': 

185 self.reduced_speed_nodes[(x, y)] = grass_speed 

186 elif self.name == 'water': 

187 self.reduced_speed_nodes[(x, y)] = water_speed 

188 else: 

189 self.blocked_nodes.add((x, y))

191 def find_available_position(self, available_nodes: Set[tuple]) -> None: 

192 """Keep looking for new coords until the object no longer 

193 touches another object. 

194 """ 

195 colliding = True 

196 if self.all_obj_attributes: 

197 while(colliding and available_nodes): 

198 for attributes in self.all_obj_attributes: 

199 # The relative position of the new object compared 

200 # to other objects already created. 

201 rules = [ 

202 (self.x + self.width) < attributes['x'], 

203 self.x > (attributes['x'] + attributes['width']), 

204 (self.y + self.height) < attributes['y'], 

205 self.y > (attributes['y'] + attributes['height']) 

206 ] 

207 # This means we are not colliding with anything. 

208 if any(rules): 

209 failed = False 

210 else: # Colliding: choose a new x,y pair. 

211 failed = True 

212 available_nodes.remove(self.xy) 

213 if not available_nodes: 

214 break 

215 self.xy = random_xy(available_nodes) 

216 self.x, self.y = self.xy 

217 break 

218 

219 colliding = False if not failed else True
